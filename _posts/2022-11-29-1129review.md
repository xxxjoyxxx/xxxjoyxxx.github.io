#### 입력 처리하는 jsp 파일 만들기
```jsp
<%@ page language="java" contentType="text/html; charset=UTF-8"
    pageEncoding="UTF-8"%>
<!-- input_ok.jsp -->
<%@page import="java.sql.*" %>
<%
	//DB 연결
	 Class.forName("com.mysql.jdbc.Driver"); 
	 String db="jdbc:mysql://localhost:3306/mydb";
	 Connection conn=DriverManager.getConnection(db,"root","1234");
	 
	 // 폼에 입력된 값 가져오기
	 request.setCharacterEncoding("utf-8");
	 String code=request.getParameter("code");
	 String name=request.getParameter("name");
	 int price=Integer.parseInt(request.getParameter("price"));
	 int yang=Integer.parseInt(request.getParameter("yang"));
	 int yul=Integer.parseInt(request.getParameter("yul"));
	 
	 // 쿼리문 작성
	 String sql="insert into product(code,name,price,yang,yul) values(?,?,?,?,?)";
	 
	 // 심부름꾼 생성
	 PreparedStatement pstmt=conn.prepareStatement(sql);
	 pstmt.setString(1, code);
	 pstmt.setString(2, name);
	 pstmt.setInt(3, price);
	 pstmt.setInt(4, yang);
	 pstmt.setInt(5, yul);
	 
	 // 쿼리문 실행
	 pstmt.executeUpdate();
	 
	 // close()
	 pstmt.close();
	 conn.close();
	 
	 // response 객체
	 response.sendRedirect("list.jsp");
	 // 해당문서로 이동
%>
```

#### 수정할 폼 만들기

#### 수정 처리하는 jsp파일
```jsp
<%@ page language="java" contentType="text/html; charset=UTF-8"
    pageEncoding="UTF-8"%>
	<!-- update_ok.jsp -->
<%@page import="java.sql.*" %>
<%
	// DB연결
	Class.forName("com.mysql.jdbc.Driver");
	String db="jdbc:mysql://localhost:3306/mydb";
	Connection conn=DriverManager.getConnection(db,"root","1234");
	
	// request 값 읽어오기
	request.setCharacterEncoding("utf-8");
	String code=request.getParameter("code");
	String name=request.getParameter("name");
	int price=Integer.parseInt(request.getParameter("price"));
	int yang=Integer.parseInt(request.getParameter("yang"));
	int yul=Integer.parseInt(request.getParameter("yul"));
	
	// 쿼리 생성
	String sql="update product set name=?,price=?,yang=?,yul=? where code=?";
	
	// 심부름꾼 생성
	PreparedStatement pstmt=conn.prepareStatement(sql);
	pstmt.setString(1, name);
	pstmt.setInt(2, price);
	pstmt.setInt(3, yang);
	pstmt.setInt(4, yul);
	pstmt.setString(5,code);
	
	// 쿼리 실행
	pstmt.executeUpdate();
	
	// close
	pstmt.close();
	conn.close();
	
	// 이동
	response.sendRedirect("list.jsp");
	
%>
```
  
  
#### 삭제 처리하는 jsp파일
```jsp
<%@ page language="java" contentType="text/html; charset=UTF-8"
    pageEncoding="UTF-8"%>
<!-- pro_delete.jsp -->
 <%@page import="java.sql.*" %>
 <%
	 // 삭제
	 
	 // DB 연결
	 Class.forName("com.mysql.jdbc.Driver"); 
	 String db="jdbc:mysql://localhost:3306/mydb";
	 Connection conn=DriverManager.getConnection(db,"root","1234");
	 
	 // request값 읽기
	 String code=request.getParameter("code");
	 
	 // 쿼리문 작성
	 String sql="delete from product where code=?";
	 // 조건이 없는 delete는 모든 레코드를 삭제한다
	 
	 // 심부름꾼 생성
	 PreparedStatement pstmt=conn.prepareStatement(sql);
	 pstmt.setString(1, code);
	 
	 // 쿼리 실행
	 pstmt.executeUpdate();
	 
	 // close()
	 pstmt.close();
	 conn.close();
	 
	 // 리스트로 이동
	 response.sendRedirect("list.jsp");
 %>
```
