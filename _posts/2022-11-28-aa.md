#### 아까 만든 입력폼을 처리할 jsp파일 만들기
  
```jsp
<%@ page language="java" contentType="text/html; charset=UTF-8"
    pageEncoding="UTF-8"%>
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>Insert title here</title>
</head>
<body>	<!-- input_ok.jsp -->
	<%@page import="java.sql.*" %>
	<%
		// DB연결
		Class.forName("com.mysql.jdbc.Driver"); 
	 	String db="jdbc:mysql://localhost:3306/mydb";
	 	Connection conn=DriverManager.getConnection(db,"root","1234");
	 	
		// request값 읽기
		int age=Integer.parseInt(request.getParameter("age"));
		int area=Integer.parseInt(request.getParameter("area"));
		String[] hobby=request.getParameterValues("hobby");
		String hob="";
		for(int i=0;i<hobby.length;i++)
		{
			hob=hob+hobby[i]+",";
		}
		
		// 쿼리문 작성
		String sql="insert into mytable(age,hobby,area) ";
		sql=sql+" values(?,?,?)";
		
		// 심부름꾼 생성
		PreparedStatement pstmt=conn.prepareStatement(sql);
		pstmt.setInt(1, age);
		pstmt.setString(2, hob);
		pstmt.setInt(3, area);
		
		// 쿼리 실행
		pstmt.executeUpdate();
		
		// close
		pstmt.close();
		conn.close();
		
		// 다른 문서로 이동
		response.sendRedirect("my_list.jsp");
		
	%>
</body>
</html>
```

#### 업데이트 폼
  
```jsp
<%@ page language="java" contentType="text/html; charset=UTF-8"
    pageEncoding="UTF-8"%>
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">

</head>

		// 수정하고자 하는 레코드를 하나 읽어온다
		// DB연결
		Class.forName("com.mysql.jdbc.Driver"); 
		String db="jdbc:mysql://localhost:3306/mydb";
		Connection conn=DriverManager.getConnection(db,"root","1234");

		String id=request.getParameter("id");

		PreparedStatement pstmt=conn.prepareStatement(sql);
		pstmt.setString(1, id);
		
		// 쿼리 실행
		ResultSet rs=pstmt.executeQuery();
		rs.next();
		
		// 읽어온 값을 폼태그에 표시
		
	%>
	<script>
		function init()
		{
			<%-- var age=<%=rs.getInt("age")%>; --%>
			var hobby="<%=rs.getString("hobby")%>";

			/* document.pkc.age[age].checked=true; */
```

#### 업데이트 처리하는 jsp파일
  
```jsp
<%@ page language="java" contentType="text/html; charset=UTF-8"
    pageEncoding="UTF-8"%>
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>Insert title here</title>
</head>
<body>	<!-- update_ok.jsp -->
	<%@page import="java.sql.*" %>
	<%
		// DB 연결
		Class.forName("com.mysql.jdbc.Driver"); 
		String db="jdbc:mysql://localhost:3306/mydb";
		Connection conn=DriverManager.getConnection(db,"root","1234");
	
		// request 값 읽기
		String id=request.getParameter("id");
		int age=Integer.parseInt(request.getParameter("age"));
		int area=Integer.parseInt(request.getParameter("area"));
		String[] hobby=request.getParameterValues("hobby");
		String hob="";
		for(int i=0;i<hobby.length;i++)
		{
			hob=hob+hobby[i]+",";
		}
		
		// 쿼리문 작성
		String sql="update mytable set age=?,hobby=?,area=? where id=?";
		
		// 심부름꾼 생성
		PreparedStatement pstmt=conn.prepareStatement(sql);
		pstmt.setInt(1, age);
		pstmt.setString(2, hob);
		pstmt.setInt(3, area);
		pstmt.setString(4, id);
		
		// 쿼리 실행
		pstmt.executeUpdate();
		
		// close
		pstmt.close();
		conn.close();
		
		// response
		response.sendRedirect("my_list.jsp");
	%>
</body>
</html>
```
  
  
#### 입력한 값을 볼 수 있는 목록 만들기
  
```jsp
<%@ page language="java" contentType="text/html; charset=UTF-8"
    pageEncoding="UTF-8"%>
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>Insert title here</title>
</head>
<body>
	<%@page import="java.sql.*" %>
	<%
	// DB연결
	Class.forName("com.mysql.jdbc.Driver"); 
	String db="jdbc:mysql://localhost:3306/mydb";
	Connection conn=DriverManager.getConnection(db,"root","1234");
	
	// 쿼리문 작성
	String sql="select * from mytable";
	
	// 심부름꾼 작성
	PreparedStatement pstmt=conn.prepareStatement(sql);
	
	// 쿼리 실행
	ResultSet rs=pstmt.executeQuery();
	
	
	%> 
	
	<table width="500" align="center" border="1">
		<tr>
			<td>나이</td>
			<td>취미</td>
			<td>지역</td>
			<td>수정</td>
			<td>삭제</td>
		</tr>
	<%
		
		while(rs.next())
		{
			String age2;
			int age=rs.getInt("age");
			switch(age)
			{
				case 0: age2="10대"; break;
				case 1: age2="20대"; break;
				case 2: age2="30대"; break;
				case 3: age2="40대"; break;
				case 4: age2="50대"; break;
				default: age2="오류";
			}
			
			String hobby2="";
			String[] hobby=rs.getString("hobby").split(",");
			for(int i=0;i<hobby.length;i++)
			{
				switch(hobby[i])
				{
					case "0": hobby2=hobby2+"낚시 "; break;
					case "1": hobby2=hobby2+"여행 "; break;
					case "2": hobby2=hobby2+"독서 "; break;
					case "3": hobby2=hobby2+"음주 "; break;
					case "4": hobby2=hobby2+"게임 "; break;
					case "5": hobby2=hobby2+"불멍 "; break;
					case "6": hobby2=hobby2+"잠자기 "; break;
					default: hobby2="오류";
				}
			}
			
			String area2;
			int area=rs.getInt("area");
			switch(area)
			{
				case 1: area2="경기도"; break;
				case 2: area2="제주도"; break;
				case 3: area2="울릉도"; break;
				case 4: area2="독도"; break;
				case 5: area2="영도"; break;
				default: area2="오류";
			}
	%>	
		<tr>
			<td><%=age2%></td>
			<td><%=hobby2%></td>
			<td><%=area2%></td>
			<td><a href="update.jsp?id=<%=rs.getInt("id")%>">수정</a></td>
			<td><a href="delete.jsp?id=<%=rs.getInt("id")%>">삭제</a></td>
		</tr>
	<%
		}	// while문 종료
	%>
		<tr>
			<td colspan="5" align="center"><a href="input.jsp">입력</a></td>
		</tr>
	</table>	
	
	<%
	// close
		rs.close();
		pstmt.close();
		conn.close();
	%>
</body>
</html>
```
