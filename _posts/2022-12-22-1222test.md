public class Main {

	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int C=sc.nextInt();
		int[] test;
		
		for(int i=0;i<C;i++)
		{
			int N=sc.nextInt();
			test=new int[N];
			double tot=0;
			
			for(int j=0;j<N;j++)
			{
				int sung=sc.nextInt();
				test[j]=sung;
				tot+=sung;
			}
			double avg=tot/N;
			double ans=0;
			
			for(int k=0;k<N;k++)
			{
				if(test[k]>avg)
				{
					ans++;
				}
			}
			System.out.printf("%.3f%%\n",(ans/N)*100);
		}

	}

}
